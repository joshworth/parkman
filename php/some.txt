        String sql = "INSERT INTO veh_receipts(veh_id,amount, txn_no, txn_type, user_id,txn_time)  VALUES (?,?,?,?,?,?)";
        //compile the statement and start a transaction
        SQLiteStatement statement = mDatabase.compileStatement(sql);
        statement.clearBindings();

        //String txnNo = Universe.BASE_SERIAL++ + Utilities.makeMinuteStamp();
        //rec.setTxnId(txnNo);

        statement.bindLong(1, rec.getVehId());
        statement.bindDouble(2, rec.getAmount());
        statement.bindString(3, rec.getTxnId());
        statement.bindString(4, rec.getTxnType());
        statement.bindLong(5, rec.getUserId());
        statement.bindLong(6, rec.getTxnTime());

        mDatabase.beginTransaction();

        statement.execute();

        //update receipt in details
        String sqlr = "UPDATE veh_details SET charge=?,exemption=?, rcno=?,rctime=? WHERE id=?";

        statement = mDatabase.compileStatement(sqlr);
        statement.clearBindings();

        statement.bindDouble(1,rec.getAmount());
        statement.bindDouble(2,rec.getExemption());
        statement.bindString(3,rec.getTxnId());
        statement.bindLong(4,rec.getTxnTime());
        statement.bindLong(5,rec.getVehId());

        statement.execute();

        mDatabase.setTransactionSuccessful();
        mDatabase.endTransaction();

    }
